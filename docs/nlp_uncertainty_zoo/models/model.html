170    def get_uncertainty(
171        self,
172        input_: torch.LongTensor,
173        metric_name: Optional[str] = None,
174        **kwargs,
175    ) -> torch.FloatTensor:
176        """
177        Get the uncertainty scores for the current batch.
178
179        Parameters
180        ----------
181        input_: torch.LongTensor
182            (Batch of) Indexed input sequences.
183        metric_name: Optional[str]
184            Name of uncertainty metric being used. If None, use metric defined under the default_uncertainty_metric
185            attribute.
186
187        Returns
188        -------
189        torch.FloatTensor
190            Uncertainty scores for the current batch.
191        """
192        if metric_name is None:
193            metric_name = self.default_uncertainty_metric
194
195        logits = self.get_logits(input_, **kwargs)
196
197        with torch.no_grad():
198            if metric_name in self.single_prediction_uncertainty_metrics:
199
200                # When model produces multiple predictions, average over them
201                if len(logits.shape) == 4:
202                    logits = logits.mean(dim=1)
203
204                return self.single_prediction_uncertainty_metrics[metric_name](logits)
205
206            elif metric_name in self.multi_prediction_uncertainty_metrics:
207                return self.multi_prediction_uncertainty_metrics[metric_name](logits)
208
209            else:
210                raise LookupError(
211                    f"Unknown metric '{metric_name}' for class '{self.__class__.__name__}'."
212                )

214    def get_num_learnable_parameters(self) -> int:
215        """
216        Return the total number of (learnable) parameters in the model.
217
218        Returns
219        -------
220        int
221            Number of learnable parameters.
222        """
223        num_parameters = 0
224
225        for param in self.parameters():
226            if param.requires_grad:
227                flattened_param = torch.flatten(param)
228                num_parameters += flattened_param.shape[0]
229
230        return num_parameters

249class Model(ABC):
250    """
251    Abstract model class. It is a wrapper that defines data loading, batching, training and evaluation loops, so that
252    the core module class can only define the model's forward pass.
253    """
254
255    def __init__(
256        self,
257        model_name: str,
258        module_class: type,
259        model_params: Dict[str, Any],
260        model_dir: Optional[str] = None,
261        device: Device = "cpu",
262    ):
263        """
264        Initialize a module.
265
266        Parameters
267        ----------
268        model_name: str
269            Name of the model.
270        module_class: type
271            Class of the model that is being wrapped.
272        model_params: Dict[str, Any]
273            Parameters to initialize the model.
274        device: Device
275            The device the model is located on.
276        """
277        self.model_name = model_name
278        self.module_class = module_class
279        self.module = module_class(**model_params, device=device)
280        self.model_dir = model_dir
281        self.model_params = model_params
282        self.device = device
283        self.loss_weights = None
284        self.to(device)
285
286        # Initialize optimizer and scheduler
287        optimizer_class = self.model_params.get("optimizer_class", optim.Adam)
288        self.optimizer = optimizer_class(
289            self.module.parameters(),
290            lr=self.model_params["lr"],
291            weight_decay=self.model_params.get("weight_decay", 0),
292        )
293
294        self.scheduler = None
295        if "scheduler_class" in self.model_params:
296            scheduler_class = self.model_params["scheduler_class"]
297            self.scheduler = scheduler_class(
298                self.optimizer, **self.model_params["scheduler_kwargs"]
299            )
300
301        # Check if model directory exists, if not, create
302        if model_dir is not None:
303            self.full_model_dir = os.path.join(model_dir, model_name)
304
305            if not os.path.exists(self.full_model_dir):
306                os.makedirs(self.full_model_dir)

308    def fit(
309        self,
310        train_split: DataLoader,
311        valid_split: Optional[DataLoader] = None,
312        verbose: bool = True,
313        weight_loss: bool = False,
314        wandb_run: Optional[WandBRun] = None,
315    ):
316        """
317        Fit the model to training data.
318
319        Parameters
320        ----------
321        train_split: DataLoader
322            Dataset the model is being trained on.
323        valid_split: Optional[DataLoader]
324            Validation set the model is being evaluated on if given.
325        verbose: bool
326            Whether to display information about current loss.
327        weight_loss: bool
328            Weight classes in loss function. Default is False.
329        wandb_run: Optional[WandBRun]
330            Weights and Biases run to track training statistics. Training and validation loss (if applicable) are
331            tracked by default, everything else is defined in _epoch_iter() and _finetune() depending on the model.
332        """
333        num_training_steps = self.model_params["num_training_steps"]
334        best_val_loss = np.inf
335        grad_clip = self.model_params.get("grad_clip", np.inf)
336        validation_interval = self.model_params["validation_interval"]
337        early_stopping_pat = self.model_params.get("early_stopping_pat", np.inf)
338        early_stopping = self.model_params.get("early_stopping", True)
339        num_no_improvements = 0
340        progress_bar = tqdm(total=num_training_steps) if verbose else None
341        best_model = dict(self.__dict__)
342
343        # Compute loss weights
344        if weight_loss:
345            counter = Counter()
346
347            for batch in train_split:
348                labels = batch["labels"]
349
350                # Flatten label lists with sum(), then filter ignore label
351                counter.update(filter(lambda label: label != -100, sum(labels.tolist(), [])))
352
353            self.loss_weights = torch.zeros(self.module.output_size, device=self.device)
354
355            for key, freq in counter.items():
356                self.loss_weights[key] = freq
357
358            del counter
359
360            self.loss_weights /= torch.sum(self.loss_weights)
361            self.loss_weights = 1 - self.loss_weights
362
363        def batch_generator(train_split: DataLoader) -> Generator[Dict[str, torch.Tensor], None, None]:
364            """
365            Quick generator that outputs batches indefinitely.
366            """
367            while True:
368                for batch in train_split:
369                    yield batch
370
371        for training_step, batch in enumerate(batch_generator(train_split)):
372
373            if training_step == num_training_steps:
374                break
375
376            self.module.train()
377
378            attention_mask, input_ids, labels = (
379                batch["attention_mask"].to(self.device),
380                batch["input_ids"].to(self.device),
381                batch["labels"].to(self.device),
382            )
383
384            batch_loss = self.get_loss(
385                input_ids,
386                labels,
387                attention_mask=attention_mask,
388                wandb_run=wandb_run,
389            )
390
391            batch_loss.backward()
392
393            clip_grad_norm_(self.module.parameters(), grad_clip)
394            self.optimizer.step()
395            self.optimizer.zero_grad(
396                set_to_none=True
397            )  # Save memory by setting to None
398
399            # Update learning rate
400            if self.scheduler is not None:
401                self.scheduler.step()
402
403                if wandb_run is not None:
404                    wandb_run.log({"batch_learning_rate": self.scheduler.get_last_lr()[0]})
405
406            batch_loss = batch_loss.cpu().detach().item()
407            if batch_loss == np.inf or np.isnan(batch_loss):
408                raise ValueError(f"Loss became NaN or inf during step {training_step + 1}.")
409
410            # Update progress bar and summary writer
411            if verbose:
412                progress_bar.set_description(
413                    f"Step {training_step + 1}: Train Loss {batch_loss:.4f}"
414                )
415                progress_bar.update(1)
416
417            if wandb_run is not None:
418                wandb_run.log({"batch_train_loss": batch_loss})
419
420            # Get validation loss
421            if valid_split is not None and training_step % validation_interval == 0 and training_step > 0:
422                self.module.eval()
423
424                with torch.no_grad():
425                    val_loss = self.eval(valid_split, wandb_run=wandb_run)
426
427                if wandb_run is not None:
428                    import wandb
429                    wandb.log({"val_loss": val_loss.item()})
430
431                if val_loss < best_val_loss:
432                    best_val_loss = val_loss.item()
433
434                    if early_stopping:
435                        best_model = dict(self.__dict__)
436
437                else:
438                    num_no_improvements += 1
439
440                    if num_no_improvements > early_stopping_pat:
441                        break
442
443        # Set current model to best model found, otherwise use last
444        if early_stopping:
445            self.__dict__ = best_model
446            del best_model
447
448        # Additional training step, e.g. temperature scaling on val
449        if valid_split is not None:
450            self._finetune(valid_split, verbose, wandb_run)
451
452        # Save model if applicable
453        if self.model_dir is not None:
454            timestamp = str(datetime.now().strftime("%d-%m-%Y_(%H:%M:%S)"))
455            torch.save(
456                self,
457                os.path.join(
458                    self.full_model_dir,
459                    f"{best_val_loss:.2f}_{timestamp}.pt",
460                ),
461                pickle_module=dill,
462                _use_new_zipfile_serialization=False,
463            )
464
465        result_dict = {
466            "model_name": self.model_name,
467            "train_loss": batch_loss,
468            "best_val_loss": best_val_loss,
469        }
470
471        return result_dict

516    def eval(self, data_split: DataLoader, wandb_run: Optional[WandBRun] = None) -> torch.Tensor:
517        """
518        Evaluate a data split.
519
520        Parameters
521        ----------
522        data_split: DataSplit
523            Data split the model should be evaluated on.
524        wandb_run: Optional[WandBRun]
525            Weights and Biases run to track training statistics. Training and validation loss (if applicable) are
526            tracked by default, everything else is defined in _epoch_iter() and _finetune() depending on the model.

528        Returns
529        -------
530        torch.Tensor
531            Loss on evaluation split.
532        """
533        self.module.eval()
534        loss = torch.zeros(1)
535
536        for batch in data_split:
537            attention_mask, input_ids, labels = (
538                batch["attention_mask"].to(self.device),
539                batch["input_ids"].to(self.device),
540                batch["labels"].to(self.device),
541            )
542            batch_loss = self.get_loss(
543                input_ids,
544                labels,
545                attention_mask=attention_mask,
546                wandb_run=wandb_run,
547            )
548
549            loss += batch_loss.detach().cpu()
550
551        self.module.train()
552
553        return loss

555    def get_loss(
556        self,
557        X: torch.Tensor,
558        y: torch.Tensor,
559        wandb_run: Optional[WandBRun] = None,
560        **kwargs,
561    ) -> torch.Tensor:
562        """
563        Get loss for a single batch. This just uses cross-entropy loss, but can be adjusted in subclasses by overwriting
564        this function.
565
566        Parameters
567        ----------
568        X: torch.Tensor
569            Batch input.
570        y: torch.Tensor
571            Batch labels.
572        wandb_run: Optional[WandBRun] = None
573            Weights and Biases run to track training statistics.
574
575        Returns
576        -------
577        torch.Tensor
578            Batch loss.
579        """
580
581        loss_function = nn.CrossEntropyLoss(
582            ignore_index=-100, weight=self.loss_weights
583        )  # Index that is used for non-masked tokens for MLM
584        preds = self.module.forward(X, **kwargs)
585
586        loss = loss_function(
587            rearrange(preds, "b t p -> (b t) p"),
588            rearrange(y, "b l -> (b l)")
589            if not self.module.is_sequence_classifier
590            else y,
591        )
592
593        return loss

310        self,
311        train_split: DataLoader,
312        valid_split: Optional[DataLoader] = None,
313        verbose: bool = True,
314        weight_loss: bool = False,
315        wandb_run: Optional[WandBRun] = None,
316    ):
317        """
318        Fit the model to training data.
319
320        Parameters
321        ----------
322        train_split: DataLoader
323            Dataset the model is being trained on.
324        valid_split: Optional[DataLoader]
325            Validation set the model is being evaluated on if given.
326        verbose: bool
327            Whether to display information about current loss.
328        weight_loss: bool
329            Weight classes in loss function. Default is False.
330        wandb_run: Optional[WandBRun]
331            Weights and Biases run to track training statistics. Training and validation loss (if applicable) are
332            tracked by default, everything else is defined in _epoch_iter() and _finetune() depending on the model.
333        """
334        num_training_steps = self.model_params["num_training_steps"]
335        best_val_loss = np.inf
336        grad_clip = self.model_params.get("grad_clip", np.inf)
337        validation_interval = self.model_params["validation_interval"]
338        early_stopping_pat = self.model_params.get("early_stopping_pat", np.inf)
339        early_stopping = self.model_params.get("early_stopping", True)
340        num_no_improvements = 0
341        progress_bar = tqdm(total=num_training_steps) if verbose else None
342        best_model = dict(self.__dict__)
343
344        # Compute loss weights
345        if weight_loss:
346            counter = Counter()
347
348            for batch in train_split:
349                labels = batch["labels"]
350
351                # Flatten label lists with sum(), then filter ignore label
352                counter.update(filter(lambda label: label != -100, sum(labels.tolist(), [])))
353
354            self.loss_weights = torch.zeros(self.module.output_size, device=self.device)
355
356            for key, freq in counter.items():
357                self.loss_weights[key] = freq
358
359            del counter
360
361            self.loss_weights /= torch.sum(self.loss_weights)
362            self.loss_weights = 1 - self.loss_weights
363
364        def batch_generator(train_split: DataLoader) -> Generator[Dict[str, torch.Tensor], None, None]:
365            """
366            Quick generator that outputs batches indefinitely.
367            """
368            while True:
369                for batch in train_split:
370                    yield batch
371
372        for training_step, batch in enumerate(batch_generator(train_split)):
373
374            if training_step == num_training_steps:
375                break
376
377            self.module.train()
378
379            attention_mask, input_ids, labels = (
380                batch["attention_mask"].to(self.device),
381                batch["input_ids"].to(self.device),
382                batch["labels"].to(self.device),
383            )
384
385            batch_loss = self.get_loss(
386                input_ids,
387                labels,
388                attention_mask=attention_mask,
389                wandb_run=wandb_run,
390            )
391
392            batch_loss.backward()
393
394            clip_grad_norm_(self.module.parameters(), grad_clip)
395            self.optimizer.step()
396            self.optimizer.zero_grad(
397                set_to_none=True
398            )  # Save memory by setting to None
399
400            # Update learning rate
401            if self.scheduler is not None:
402                self.scheduler.step()
403
404                if wandb_run is not None:
405                    wandb_run.log({"batch_learning_rate": self.scheduler.get_last_lr()[0]})
406
407            batch_loss = batch_loss.cpu().detach().item()
408            if batch_loss == np.inf or np.isnan(batch_loss):
409                raise ValueError(f"Loss became NaN or inf during step {training_step + 1}.")
410
411            # Update progress bar and summary writer
412            if verbose:
413                progress_bar.set_description(
414                    f"Step {training_step + 1}: Train Loss {batch_loss:.4f}"
415                )
416                progress_bar.update(1)
417
418            if wandb_run is not None:
419                wandb_run.log({"batch_train_loss": batch_loss})
420
421            # Get validation loss
422            if valid_split is not None and training_step % validation_interval == 0 and training_step > 0:
423                self.module.eval()
424
425                with torch.no_grad():
426                    val_loss = self.eval(valid_split, wandb_run=wandb_run)
427
428                if wandb_run is not None:
429                    import wandb
430                    wandb.log({"val_loss": val_loss.item()})
431
432                if val_loss < best_val_loss:
433                    best_val_loss = val_loss.item()
434
435                    if early_stopping:
436                        best_model = dict(self.__dict__)
437
438                else:
439                    num_no_improvements += 1
440
441                    if num_no_improvements > early_stopping_pat:
442                        break
443
444        # Set current model to best model found, otherwise use last
445        if early_stopping:
446            self.__dict__ = best_model
447            del best_model
448
449        # Additional training step, e.g. temperature scaling on val
450        if valid_split is not None:
451            self._finetune(valid_split, verbose, wandb_run)
452
453        # Save model if applicable
454        if self.model_dir is not None:
455            timestamp = str(datetime.now().strftime("%d-%m-%Y_(%H:%M:%S)"))
456            torch.save(
457                self,
458                os.path.join(
459                    self.full_model_dir,
460                    f"{best_val_loss:.2f}_{timestamp}.pt",
461                ),
462                pickle_module=dill,
463                _use_new_zipfile_serialization=False,
464            )
465
466        result_dict = {
467            "model_name": self.model_name,
468            "train_loss": batch_loss,
469            "best_val_loss": best_val_loss,
470        }
471
472        return result_dict

317        """
318        Fit the model to training data.
319
320        Parameters
321        ----------
322        train_split: DataLoader
323            Dataset the model is being trained on.
324        valid_split: Optional[DataLoader]
325            Validation set the model is being evaluated on if given.
326        verbose: bool
327            Whether to display information about current loss.
328        weight_loss: bool
329            Weight classes in loss function. Default is False.
330        wandb_run: Optional[WandBRun]
331            Weights and Biases run to track training statistics. Training and validation loss (if applicable) are
332            tracked by default, everything else is defined in _epoch_iter() and _finetune() depending on the model.
333        """
334        num_training_steps = self.model_params["num_training_steps"]
335        best_val_loss = np.inf
336        grad_clip = self.model_params.get("grad_clip", np.inf)
337        validation_interval = self.model_params["validation_interval"]
338        early_stopping_pat = self.model_params.get("early_stopping_pat", np.inf)
339        early_stopping = self.model_params.get("early_stopping", True)
340        num_no_improvements = 0
341        progress_bar = tqdm(total=num_training_steps) if verbose else None
342        best_model = dict(self.__dict__)
343
344        # Compute loss weights
345        if weight_loss:
346            counter = Counter()
347
348            for batch in train_split:
349                labels = batch["labels"]
350
351                # Flatten label lists with sum(), then filter ignore label
352                counter.update(filter(lambda label: label != -100, sum(labels.tolist(), [])))
353
354            self.loss_weights = torch.zeros(self.module.output_size, device=self.device)
355
356            for key, freq in counter.items():
357                self.loss_weights[key] = freq
358
359            del counter
360
361            self.loss_weights /= torch.sum(self.loss_weights)
362            self.loss_weights = 1 - self.loss_weights
363
364        def batch_generator(train_split: DataLoader) -> Generator[Dict[str, torch.Tensor], None, None]:
365            """
366            Quick generator that outputs batches indefinitely.
367            """
368            while True:
369                for batch in train_split:
370                    yield batch
371
372        for training_step, batch in enumerate(batch_generator(train_split)):
373
374            if training_step == num_training_steps:
375                break
376
377            self.module.train()
378
379            attention_mask, input_ids, labels = (
380                batch["attention_mask"].to(self.device),
381                batch["input_ids"].to(self.device),
382                batch["labels"].to(self.device),
383            )
384
385            batch_loss = self.get_loss(
386                input_ids,
387                labels,
388                attention_mask=attention_mask,
389                wandb_run=wandb_run,
390            )
391
392            batch_loss.backward()
393
394            clip_grad_norm_(self.module.parameters(), grad_clip)
395            self.optimizer.step()
396            self.optimizer.zero_grad(
397                set_to_none=True
398            )  # Save memory by setting to None
399
400            # Update learning rate
401            if self.scheduler is not None:
402                self.scheduler.step()
403
404                if wandb_run is not None:
405                    wandb_run.log({"batch_learning_rate": self.scheduler.get_last_lr()[0]})
406
407            batch_loss = batch_loss.cpu().detach().item()
408            if batch_loss == np.inf or np.isnan(batch_loss):
409                raise ValueError(f"Loss became NaN or inf during step {training_step + 1}.")
410
411            # Update progress bar and summary writer
412            if verbose:
413                progress_bar.set_description(
414                    f"Step {training_step + 1}: Train Loss {batch_loss:.4f}"
415                )
416                progress_bar.update(1)
417
418            if wandb_run is not None:
419                wandb_run.log({"batch_train_loss": batch_loss})
420
421            # Get validation loss
422            if valid_split is not None and training_step % validation_interval == 0 and training_step > 0:
423                self.module.eval()
424
425                with torch.no_grad():
426                    val_loss = self.eval(valid_split, wandb_run=wandb_run)
427
428                if wandb_run is not None:
429                    import wandb
430                    wandb.log({"val_loss": val_loss.item()})
431
432                if val_loss < best_val_loss:
433                    best_val_loss = val_loss.item()
434
435                    if early_stopping:
436                        best_model = dict(self.__dict__)
437
438                else:
439                    num_no_improvements += 1
440
441                    if num_no_improvements > early_stopping_pat:
442                        break
443
444        # Set current model to best model found, otherwise use last
445        if early_stopping:
446            self.__dict__ = best_model
447            del best_model
448
449        # Additional training step, e.g. temperature scaling on val
450        if valid_split is not None:
451            self._finetune(valid_split, verbose, wandb_run)
452
453        # Save model if applicable
454        if self.model_dir is not None:
455            timestamp = str(datetime.now().strftime("%d-%m-%Y_(%H:%M:%S)"))
456            torch.save(
457                self,
458                os.path.join(
459                    self.full_model_dir,
460                    f"{best_val_loss:.2f}_{timestamp}.pt",
461                ),
462                pickle_module=dill,
463                _use_new_zipfile_serialization=False,
464            )
465
466        result_dict = {
467            "model_name": self.model_name,
468            "train_loss": batch_loss,
469            "best_val_loss": best_val_loss,
470        }
471
472        return result_dict

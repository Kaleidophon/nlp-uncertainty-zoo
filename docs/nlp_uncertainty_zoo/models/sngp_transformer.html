 27class SNGPModule(nn.Module):
 28    """
 29    Spectral-normalized Gaussian Process output layer, as presented in
 30    `Liu et al. (2020) <https://arxiv.org/pdf/2006.10108.pdf>`_. Requires underlying model to contain residual
 31    connections in order to maintain bi-Lipschitz constraint.
 32    """
 33
 34    def __init__(
 35        self,
 36        hidden_size: int,
 37        output_size: int,
 38        ridge_factor: float,
 39        scaling_coefficient: float,
 40        beta_length_scale: float,
 41        kernel_amplitude: float,
 42        num_predictions: int,
 43        device: Device,
 44        **build_params,
 45    ):
 46        """
 47        Initialize a SNGP output layer.
 48
 49        Parameters
 50        ----------
 51        hidden_size: int
 52            Hidden size of last regular network layer.
 53        last_layer_size: int
 54            Size of last layer before output layer. Called D_L in the original paper.
 55        output_size: int
 56            Size of output layer, so number of classes.
 57        ridge_factor: float
 58            Factor that identity sigma hat matrices of the SNGP layer are multiplied by.
 59        scaling_coefficient: float
 60            Momentum factor that is used when updating the sigma hat matrix of the SNGP layer during the last training
 61            epoch.
 62        beta_length_scale: float
 63            Factor for the variance parameter of the normal distribution all beta parameters of the SNGP layer are
 64            initialized from.
 65        kernel_amplitude: float
 66            Kernel amplitude used when computing GP features.
 67        num_predictions: int
 68            Number of predictions sampled from the GP in the SNGP layer to come to the final prediction.
 69        device: Device
 70            Device the replication is performed on.
 71        """
 72        super().__init__()
 73        self.device = device
 74
 75        self.hidden_size = hidden_size
 76        self.output_size = output_size
 77        self.ridge_factor = ridge_factor
 78        self.scaling_coefficient = scaling_coefficient
 79        self.beta_length_scale = beta_length_scale
 80        self.kernel_amplitude = kernel_amplitude
 81        self.num_predictions = num_predictions
 82
 83        # ### Init parameters
 84
 85        # Random, frozen output layer
 86        self.output = nn.Linear(self.hidden_size, self.output_size)
 87        # Change init of weights and biases following Liu et al. (2020)
 88        self.output.weight.data.normal_(0, 0.05)
 89        self.output.bias.data.uniform_(0, 2 * math.pi)
 90
 91        # This layer is frozen right after init
 92        self.output.weight.requires_grad = False
 93        self.output.bias.requires_grad = False
 94
 95        # Bundle all beta_k vectors into a matrix
 96        self.Beta = nn.Linear(output_size, output_size)
 97        self.Beta.weight.data.normal_(0, beta_length_scale)
 98        self.Beta.bias.data = torch.zeros(output_size)
 99
100        # Initialize inverse of sigma hat, one matrix in total to save memory
101        self.sigma_hat_inv = torch.eye(output_size, device=self.device) * self.beta_length_scale
102        self.sigma_hat = torch.zeros(output_size, output_size, device=device)
103        self.inversed_sigma = False
104
105        # Multivariate normal distributions that beta columns are sampled from after inverting sigma hat
106        self.beta_dists = [None for _ in range(output_size)]
107
108    def _get_features(
109        self, x: torch.FloatTensor
110    ) -> Tuple[torch.FloatTensor, torch.FloatTensor]:
111        """
112        Get posterior mean / logits and Phi feature matrix given an input.
113
114        Parameters
115        ----------
116        x: torch.FloatTensor
117            Last hidden state of underlying model.
118
119        Returns
120        -------
121        Tuple[torch.FloatTensor, torch.FloatTensor]
122            Tensors of posterior mean and Phi matrix.
123        """
124        Phi = math.sqrt(2 * self.kernel_amplitude ** 2 / self.output_size) * torch.cos(
125            self.output(-x)
126        )  # batch_size x last_layer_size
127        # Logits: batch_size x last_layer_size @ last_layer_size x output_size -> batch_size x output_size
128        post_mean = self.Beta(Phi)
129
130        return post_mean, Phi
131
132    def forward(
133        self, x: torch.FloatTensor
134    ) -> torch.FloatTensor:
135        """
136        Forward pass for SNGP layer.
137
138        Parameters
139        ----------
140        x: torch.FloatTensor
141            Last hidden state of underlying model.
142
143        Returns
144        -------
145        torch.FloatTensor
146            Logits for the current batch.
147        """
148        logits, Phi = self._get_features(x)
149
150        if self.training:
151            with torch.no_grad():
152                probs = F.softmax(logits, dim=-1)  # batch_size x seq_len x output_size
153                max_probs = torch.max(probs, dim=-1)[0]  # batch_size x seq_len
154                max_probs = max_probs * (1 - max_probs)
155
156                Phi = Phi.unsqueeze(-1)  # Make it batch_size x last_layer_size x 1
157
158                # Vectorized version of eq. 9
159                # b: batch size
160                # s: sequence length
161                # o, p: output size
162                # z: singleton dimension
163                # Integrate multiplication with max_probs into einsum to avoid producing another 4D tensor
164                PhiPhi = torch.einsum(
165                    "bsoz,bszp->op",
166                    Phi * max_probs.unsqueeze(-1).unsqueeze(-1),
167                    torch.transpose(Phi, 2, 3)
168                )
169                self.sigma_hat_inv *= self.scaling_coefficient
170                self.sigma_hat_inv += (1 - self.scaling_coefficient) * PhiPhi
171
172        return logits
173
174    def predict(self, x: torch.FloatTensor, num_predictions: Optional[int] = None):
175        """
176        Get predictions for the current batch.
177
178        Parameters
179        ----------
180        x: torch.FloatTensor
181            Last hidden state of underlying model.
182        num_predictions: Optional[int]
183            Number of predictions sampled from the GP in the SNGP layer to come to the final prediction. If None, number
184            specified during initialization is used.
185
186        Returns
187        -------
188        torch.FloatTensor
189            Class probabilities for current batch.
190        """
191
192        logits = self.get_logits(x, num_predictions)
193
194        out = F.softmax(logits, dim=-1).mean(dim=1)
195
196        return out

198    def get_logits(self, x: torch.FloatTensor, num_predictions: Optional[int] = None):
199        """
200        Get the logits for an input. Results in a tensor of size batch_size x num_predictions x seq_len x output_size
201        depending on the model type.
202
203        Parameters
204        ----------
205        x: torch.LongTensor
206            Input to the model, containing all token IDs of the current batch.
207        num_predictions: int
208            Number of predictions sampled from the GP in the SNGP layer to come to the final prediction. If None, number
209            specified during initialization is used.
210
211        Returns
212        -------
213        torch.FloatTensor
214            Logits for the current batch.
215        """
216
217        if num_predictions is None:
218            num_predictions = self.num_predictions
219
220        # In case the Sigma matrix wasn't inverted yet, make sure that it is here.
221        # Also set inversed_sigma to False again in case this is called during validation so that the matrix will be
222        # updated over the training time again.
223        if not self.inversed_sigma:
224            self.invert_sigma_hat()
225            self.inversed_sigma = False
226
227        if num_predictions is None:
228            num_predictions = self.num_predictions
229
230        _, Phi = self._get_features(x)
231
232        # Sample num_predictions Beta matrices
233        beta_samples = torch.stack(
234            [
235                dist.rsample((num_predictions, ))
236                for dist in self.beta_dists
237            ],
238            dim=-1
239        )
240
241        # Because we just stacked num_predictions samples for every column of the beta matrix, we now have to switch
242        # the last two dimensions to obtain a num_predictions Beta matrices
243        beta_samples = rearrange(beta_samples, "n c r -> n r c")
244
245        # Now compute different logits using betas sampled from the Laplace posterior. This operation is a batch
246        # instance-wise time step-wise multiplication of features with one Beta matrix per num_predictions.
247        # b: batch
248        # s: sequence length
249        # k: number of classes
250        # n: number of predictions
251        logits = torch.einsum("bsk,nkk->bnsk", Phi, beta_samples)
252
253        return logits
254
255    def invert_sigma_hat(self) -> None:
256        """
257        Invert the sigma hat matrix.
258        """
259        try:
260            self.sigma_hat = torch.inverse(self.sigma_hat_inv)
261
262        except RuntimeError:
263            warnings.warn(f"Matrix could not be inverted, compute pseudo-inverse instead.")
264            self.sigma_hat = torch.linalg.pinv(self.sigma_hat_inv)
265
266        self.inversed_sigma = True
267
268        # Create multivariate normal distributions to sample columns from Beta matrix from. Updated every time
269        # sigma_hat is updated since it sigma_hat is the covariance matrix used.
270        self.beta_dists = [
271            torch.distributions.multivariate_normal.MultivariateNormal(
272                self.Beta.weight.data[:, k], covariance_matrix=self.sigma_hat,
273            )
274            for k in range(self.output_size)
275        ]

107        self.beta_dists = [None for _ in range(output_size)]

109    def _get_features(
110        self, x: torch.FloatTensor

113        Get posterior mean / logits and Phi feature matrix given an input.
114
115        Parameters
116        ----------
117        x: torch.FloatTensor
118            Last hidden state of underlying model.
119
120        Returns
121        -------
122        Tuple[torch.FloatTensor, torch.FloatTensor]
123            Tensors of posterior mean and Phi matrix.
124        """
125        Phi = math.sqrt(2 * self.kernel_amplitude ** 2 / self.output_size) * torch.cos(
126            self.output(-x)
127        )  # batch_size x last_layer_size
128        # Logits: batch_size x last_layer_size @ last_layer_size x output_size -> batch_size x output_size
129        post_mean = self.Beta(Phi)
130
131        return post_mean, Phi

140        ----------

142            Last hidden state of underlying model.

153                probs = F.softmax(logits, dim=-1)  # batch_size x seq_len x output_size

216        """

218        if num_predictions is None:

220
221        # In case the Sigma matrix wasn't inverted yet, make sure that it is here.
222        # Also set inversed_sigma to False again in case this is called during validation so that the matrix will be
223        # updated over the training time again.
224        if not self.inversed_sigma:
225            self.invert_sigma_hat()
226            self.inversed_sigma = False

235            [
236                dist.rsample((num_predictions, ))
237                for dist in self.beta_dists
238            ],
239            dim=-1
240        )
241
242        # Because we just stacked num_predictions samples for every column of the beta matrix, we now have to switch
243        # the last two dimensions to obtain a num_predictions Beta matrices
244        beta_samples = rearrange(beta_samples, "n c r -> n r c")
245
246        # Now compute different logits using betas sampled from the Laplace posterior. This operation is a batch
247        # instance-wise time step-wise multiplication of features with one Beta matrix per num_predictions.
248        # b: batch
249        # s: sequence length
250        # k: number of classes
251        # n: number of predictions
252        logits = torch.einsum("bsk,nkk->bnsk", Phi, beta_samples)
253
254        return logits

259        """

261            self.sigma_hat = torch.inverse(self.sigma_hat_inv)
262

270        # sigma_hat is updated since it sigma_hat is the covariance matrix used.
271        self.beta_dists = [
272            torch.distributions.multivariate_normal.MultivariateNormal(

274            )
275            for k in range(self.output_size)

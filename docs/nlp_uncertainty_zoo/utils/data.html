 29class ModifiedDataCollatorForLanguageModeling(DataCollatorForLanguageModeling):
 30    """
 31    Modified version of the DataCollatorForLanguageModelling. The only change introduced is to the __call__ function,
 32    where an offset between input_ids and labels for next token prediction language modelling in order to be consistent
 33    with the rest of the code base.
 34    """
 35
 36    tokenizer: PreTrainedTokenizerBase
 37    mlm: bool = True
 38    mlm_probability: float = 0.15
 39    pad_to_multiple_of: Optional[int] = None
 40
 41    def __call__(
 42        self, examples: List[Union[List[int], torch.Tensor, Dict[str, torch.Tensor]]]
 43    ) -> Dict[str, torch.Tensor]:
 44        # Handle dict or lists with proper padding and conversion to tensor.
 45        if isinstance(examples[0], (dict, BatchEncoding)):
 46            batch = self.tokenizer.pad(
 47                examples,
 48                return_tensors="pt",
 49                pad_to_multiple_of=self.pad_to_multiple_of,
 50            )
 51        else:
 52            batch = {
 53                "input_ids": _collate_batch(
 54                    examples, self.tokenizer, pad_to_multiple_of=self.pad_to_multiple_of
 55                )
 56            }
 57
 58        # If special token mask has been preprocessed, pop it from the dict.
 59        special_tokens_mask = batch.pop("special_tokens_mask", None)
 60        if self.mlm:
 61            batch["input_ids"], batch["labels"] = self.mask_tokens(
 62                batch["input_ids"], special_tokens_mask=special_tokens_mask
 63            )
 64        else:
 65            labels = batch["input_ids"].clone()
 66
 67            # ### Custom Modification ###
 68            labels = labels[:, 1:]
 69            batch["input_ids"] = batch["input_ids"][:, :-1]
 70            batch["attention_mask"] = batch["attention_mask"][:, :-1]
 71            # ### Custom Modifcation End ###
 72
 73            if self.tokenizer.pad_token_id is not None:
 74                labels[labels == self.tokenizer.pad_token_id] = -100
 75            batch["labels"] = labels
 76        return batch

 79class DatasetBuilder(ABC):
 80    """
 81    Abstract dataset builder class used to create a variety of different dataset types, including sequence prediction,
 82    token prediction, next-token-prediction language modelling and masked language modelling.
 83    """
 84
 85    def __init__(
 86        self,
 87        name: str,
 88        data_dir: str,
 89        splits: Dict[str, Any],
 90        type_: str,
 91        tokenizer: PreTrainedTokenizerBase,
 92        max_length: int,
 93        sampler_class: Optional[Type] = None,
 94        sampler_kwargs: Optional[SamplerKwargs] = None,
 95        num_jobs: Optional[int] = 1,
 96    ):
 97        """
 98        Initialize a DatasetBuilder.
 99
100        Parameters
101        ----------
102        name: str
103            Name of the final dataset.
104        data_dir: str
105            Directory in which data splits are located.
106        splits: Dict[str, Any]
107            Dictionary pointing to the files containing the training, validation and test split.
108        type_: str
109            String that further specifies the type of dataset being built (see LanguageModellingDatasetBuilder and
110            ClassificationDatasetBuilder for more detail).
111        tokenizer: PreTrainedTokenizerBase
112            Pre-trained tokenizer.
113        max_length: int
114            Maximum sequence length.
115        sampler_class: Optional[Sampler]
116            Sampler class used to (sub-)sample the dataset.
117        sampler_kwargs: Optional[SamplerKwargs]
118            Keyword arguments to initialize the sampler.
119        num_jobs: Optional[int]
120            Number of jobs used to build the dataset (on CPU). Default is 1.
121        """
122        self.name = name
123        self.data_dir = data_dir
124        self.splits = splits
125        self.type = type_
126        self.tokenizer = tokenizer
127        self.max_length = max_length
128        self.sampler_class = sampler_class
129        self.sampler_kwargs = sampler_kwargs
130        self.num_jobs = num_jobs
131        self.label_encoder = None
132
133        # To be set after calling build()
134        self.dataset = None
135        self.dataloaders = None
136
137    @abstractmethod
138    def build(
139        self, batch_size: int, **dataloader_kwargs: Dict[str, Any]
140    ) -> Dict[str, DataLoader]:
141        """
142        Build a dataset.
143
144        Parameters
145        ----------
146        batch_size: int
147            The desired batch size.
148
149        Returns
150        -------
151        Dict[str, DataLoader]
152            Dictionary of DataLoaders for every given split.
153        """
154        pass

157class LanguageModellingDatasetBuilder(DatasetBuilder):
158    """
159    DatasetBuilder for language modelling datasets. This includes "classic" language modelling (aka next token
160    prediction) as well as masked language modelling.
161    """
162
163    def build(
164        self, batch_size: int, **dataloader_kwargs: Dict[str, Any]
165    ) -> Dict[str, DataLoader]:
166        """
167        Build a language modelling dataset.
168
169        Parameters
170        ----------
171        batch_size: int
172            The desired batch size.
173
174        Returns
175        -------
176        Dict[str, DataLoader]
177            Dictionary of DataLoaders for every given split.
178        """
179        assert self.type in [
180            "mlm",
181            "next_token_prediction",
182        ], f"Invalid type '{self.type}' found, must be mlm or next_token_prediction."
183
184        self.dataset = load_dataset("text", data_files=self.splits)
185
186        # The following is basically copied from the corresponding HuggingFace tutorial: https://youtu.be/8PmhEIXhBvI
187        self.dataset = self.dataset.map(
188            lambda inst: self.tokenizer(
189                inst["text"],
190                truncation=True,
191                padding="max_length",
192                max_length=self.max_length,
193            ),
194            batched=True,
195            num_proc=self.num_jobs,
196        )
197        self.dataset.set_format(type="torch", columns=["input_ids", "attention_mask"])
198
199        mlm_kwargs = (
200            {"mlm": True, "mlm_probability": 0.15}
201            if self.type == "mlm"
202            else {"mlm": False}
203        )
204
205        # Note: Collater is a callable object used as the Pytorch DataLoader collate_fn arg
206        collator = ModifiedDataCollatorForLanguageModeling(self.tokenizer, **mlm_kwargs)
207
208        # Create the DataLoader for every split
209        self.dataloaders = {
210            split: DataLoader(
211                self.dataset[split],
212                batch_size=batch_size,
213                collate_fn=collator,
214                sampler=self.sampler_class(
215                    data_source=self.dataset[split],
216                    **self.sampler_kwargs[split],
217                )
218                if self.sampler_class is not None and split in self.sampler_kwargs
219                else None,
220                **dataloader_kwargs,
221            )
222            for split in self.splits
223        }
224
225        return self.dataloaders

228class ClassificationDatasetBuilder(DatasetBuilder):
229    """
230    DatasetBuilder for classification datasets. This includes sequence classification and token classification /
231    sequence labelling.
232    """
233
234    def build(
235        self, batch_size: int, **dataloader_kwargs: Dict[str, Any]
236    ) -> Dict[str, DataLoader]:
237        """
238        Build a language modelling dataset.
239
240        Parameters
241        ----------
242        batch_size: int
243            The desired batch size.
244
245        Returns
246        -------
247        Dict[str, DataLoader]
248            Dictionary of DataLoaders for every given split.
249        """
250        assert self.type in [
251            "sequence_classification",
252            "token_classification",
253        ], f"Invalid type '{self.type}' found, must be sequence_classification or token_classification."
254
255        self.dataset = load_dataset(
256            "csv",
257            data_files=self.splits,
258            delimiter="\t",
259            column_names=["sentence", "labels"],
260        )
261
262        # Extract all classes from data
263        classes, labeling_func = None, None
264        if self.type == "sequence_classification":
265            # This one-liner goes through all the labels occuring in the different splits and adds them to a set
266            classes = reduce(
267                lambda x, y: set(x).union(y),
268                [self.dataset[split]["labels"] for split in self.splits],
269            )
270            labeling_func = lambda inst: {
271                "labels": self.label_encoder.transform([inst["labels"]])[0]
272            }
273
274        elif self.type == "token_classification":
275            classes = reduce(
276                lambda x, y: set(x).union(y),
277                [
278                    labels.split(" ")
279                    for split in self.splits
280                    for labels in self.dataset[split]["labels"]
281                ],
282            )
283            labeling_func = lambda inst: {
284                "labels": self.label_encoder.transform(inst["labels"].split(" "))
285            }
286
287        # Encode classes
288        self.label_encoder = LabelEncoder()
289        self.label_encoder.fit(list(classes))
290
291        # Replace with classes with labels
292        self.dataset = self.dataset.map(
293            labeling_func,
294            batched=False,
295            with_indices=False,
296            num_proc=self.num_jobs,
297        )
298
299        # The following is basically copied from the corresponding HuggingFace tutorial: https://youtu.be/8PmhEIXhBvI
300        if self.type == "sequence_classification":
301            self.dataset = self.dataset.map(
302                lambda inst: self.tokenizer(
303                    inst["sentence"],
304                    truncation=True,
305                    padding="max_length",
306                    max_length=self.max_length,
307                ),
308                batched=True,
309                num_proc=self.num_jobs,
310            )
311
312        # For token classification, we have to adjust the token labels according to the subwords that the tokenizer
313        # creates. Unfortunately, we have to do that manually and the code is quite ugly. Sorry!
314        elif self.type == "token_classification":
315
316            def create_inst_data(inst: Dict[str, Any]) -> Dict[str, Any]:
317                # Dummy object, see explanations below
318                class LabelledInfo:
319                    val = False
320
321                sentence, labels = inst["sentence"], inst["labels"]
322
323                # Create data structure that maps the index of a character belonging to a token to the tokens label.
324                # Also, create second structure mapping from character index to bool indicating whether token this
325                # character belongs to was already labeled once - this way, splitting one labelled token into
326                # multiple sub-word tokens later only labels the first ones and gives the ignore label -100 to the
327                # other ones.
328                char_idx2label = defaultdict(lambda: -100)
329                char_idx2relabeled = defaultdict(lambda: LabelledInfo())
330                char_idx = 0
331                for token_idx, token in enumerate(sentence.split(" ")):
332                    # Dirty trick: Assign the same INSTANCE of a bool value wrapped in an object to all char indices
333                    # of a sub-word such that changing one of them changes all of them, since they all point to the
334                    # same object in memory. I know this is ugly and confusing! I also don't like this!
335                    # Don't you think that I also know that this is insane??
336                    inf = LabelledInfo()
337                    char_idx2relabeled.update(
338                        {i: inf for i in range(char_idx + 1, char_idx + len(token) + 1)}
339                    )
340
341                    # Update the dict mapping from char index to label
342                    char_idx2label.update(
343                        {
344                            # The min() expression is an ugly fix for cases in which a tokenizer butchers unusual
345                            # expressions like an emoticon at the end of the sequence, suddenly creating a missing
346                            # label. In this case, just repeat the last sequence label.
347                            i: labels[min(token_idx, len(labels) - 1)]
348                            for i in range(char_idx + 1, char_idx + len(token) + 1)
349                        }
350                    )
351
352                    char_idx += len(token) + 1  # Account for whitespace
353
354                inst_encoding = self.tokenizer(
355                    sentence,
356                    truncation=True,
357                    padding="max_length",
358                    max_length=self.max_length,
359                    return_offsets_mapping=True,
360                )
361
362                adjusted_labels = []  # New labels for sub-word tokens go here
363                # Here, we have one single offset mapping per sub-word token, indicating which characters it span in
364                # original sentence.
365                for _, char_end in inst_encoding["offset_mapping"]:
366                    # Check if one of the sub-word tokens belong to a word has been labelled already
367                    if not char_idx2relabeled[char_end].val:
368                        adjusted_labels.append(char_idx2label[char_end])
369                        char_idx2relabeled[char_end].val = True
370
371                    else:
372                        adjusted_labels.append(-100)
373
374                # Update the instance info
375                return {**inst_encoding.data, "labels": adjusted_labels}
376
377            # Finally map that shit function over the dataset
378            self.dataset = self.dataset.map(
379                create_inst_data,
380                batched=False,
381                num_proc=self.num_jobs,
382            )
383
384        self.dataset.set_format(
385            type="torch", columns=["input_ids", "attention_mask", "labels"]
386        )
387
388        # Create the DataLoader for every split
389        self.dataloaders = {
390            split: DataLoader(
391                self.dataset[split],
392                batch_size=batch_size,
393                sampler=self.sampler_class(
394                    data_source=self.dataset[split],
395                    **self.sampler_kwargs[split],
396                )
397                if self.sampler_class is not None and split in self.sampler_kwargs
398                else None,
399                **dataloader_kwargs,
400            )
401            for split in self.splits
402        }
403
404        return self.dataloaders

332                for token_idx, token in enumerate(sentence.split(" ")):
333                    # Dirty trick: Assign the same INSTANCE of a bool value wrapped in an object to all char indices
334                    # of a sub-word such that changing one of them changes all of them, since they all point to the

346                            # expressions like an emoticon at the end of the sequence, suddenly creating a missing

356                    sentence,
357                    truncation=True,
358                    padding="max_length",
359                    max_length=self.max_length,
360                    return_offsets_mapping=True,
361                )
362
363                adjusted_labels = []  # New labels for sub-word tokens go here
364                # Here, we have one single offset mapping per sub-word token, indicating which characters it span in
365                # original sentence.
366                for _, char_end in inst_encoding["offset_mapping"]:
367                    # Check if one of the sub-word tokens belong to a word has been labelled already
368                    if not char_idx2relabeled[char_end].val:
369                        adjusted_labels.append(char_idx2label[char_end])
370                        char_idx2relabeled[char_end].val = True
371
372                    else:
373                        adjusted_labels.append(-100)
374
375                # Update the instance info
376                return {**inst_encoding.data, "labels": adjusted_labels}
377

379            self.dataset = self.dataset.map(
380                create_inst_data,

 21def create_probs_from_dict(
 22    freq_dict: Dict[int, int], max_label: Optional[int] = None
 23) -> np.array:
 24    """
 25    Auxiliary function creating a numpy array containing a categorical distribution over integers from
 26    a dictionary of frequencies.
 27
 28    Parameters
 29    ----------
 30    freq_dict: Dict[int, int]
 31        Dictionary mapping from class labels to frequencies.
 32    max_label: Optional[int]
 33        Maximum value of a class label aka number of classes (minus 1). If None, tyis is based on the maximum valued
 34        key in freq_dict.
 35
 36    Returns
 37    -------
 38    np.array
 39        Distribution over class labels as a numpy array.
 40    """
 41    if max_label is None:
 42        max_label = max(freq_dict.keys())
 43
 44    probs = np.zeros(max_label + 1)
 45
 46    for key, freq in freq_dict.items():
 47        probs[key] = freq
 48
 49    probs /= sum(probs)
 50
 51    return probs

 54def merge_freq_dicts(
 55    freqs_a: Dict[int, int], freqs_b: Dict[int, int]
 56) -> Dict[int, int]:
 57    """
 58    Merge two dictionaries of frequencies. Used for creating data statistics before sub-sampling, where statistics for
 59    each instance are collected via different jobs and then merged.
 60
 61    Parameters
 62    ----------
 63    freqs_a: Dict[int, int]
 64        First frequency dictionary.
 65    freqs_b: Dict[int, int]
 66        Second frequency dictionary.
 67
 68    Returns
 69    -------
 70    Dict[int, int]
 71        New frequency dictionary.
 72    """
 73    return {
 74        key: freqs_a.get(key, 0) + freqs_b.get(key, 0)
 75        for key in (set(freqs_a) | set(freqs_b))
 76    }

 79def merge_instance_dicts(
 80    lengths_a: Dict[int, List[int]], lengths_b: Dict[int, List[int]]
 81) -> Dict[int, List[int]]:
 82    """
 83    Merge two dictionaries of instances lights, where inputs are grouped by a common characteristic, e.g. length. Used
 84    for creating data statistics before sub-sampling, where statistics for each instance are collected via different
 85    jobs and then merged.
 86
 87    Parameters
 88    ----------
 89    lengths_a: Dict[int, List[int]]
 90        First instance dictionary.
 91    lengths_b: Dict[int, List[int]]
 92        Second instance dictionary.
 93
 94    Returns
 95    -------
 96    Dict[int, List[int]]
 97         New instance dictionary.
 98    """
 99    return {
100        key: lengths_a.get(key, []) + lengths_b.get(key, [])
101        for key in (set(lengths_a) | set(lengths_b))
102    }

108class Subsampler(Sampler, ABC):
109    """
110    Abstract base class of any sampler that sub-samples a dataset to a given target size.
111    """
112
113    def __init__(
114        self,
115        data_source: Sized,
116        target_size: int,
117        num_jobs: int = 1,
118        seed: Optional[int] = None,
119    ):
120        """
121        Initialize a sub-sampler.
122
123        Parameters
124        ----------
125        data_source: Sized
126            Iterable of data corresponding to a split. Usually is a list of dicts, containing input ids, attention masks
127            and labels for each instance.
128        target_size: int
129            Number of instances that should be contained in the sub-sampled data set.
130        num_jobs: int
131            Number of jobs used to process data before sampling.
132        seed: Optional[int]
133            Seed set for reproducibility.
134        """
135        super().__init__(data_source)
136        self.target_size = target_size
137        self.num_jobs = num_jobs
138        self.seed = seed
139        self.indices = None
140        self._create_indices(data_source)
141
142    @abstractmethod
143    def _create_indices(self, data_source: Sized):
144        """
145        Analyze the given data in order to determining the sampling strategy.
146
147        Parameters
148        ----------
149        data_source: Sized
150            Data containing instances of data split.
151        """
152        pass
153
154    def __iter__(self):
155        return iter(self.indices)
156
157    def __len__(self):
158        return len(self.indices)

161class LanguageModellingSampler(Subsampler):
162    """
163    Sampler specific to language modelling. The sub-sampling strategy here is to approximately maintain the same
164    distribution of sentence lengths as in the original corpus, and to maintain contiguous paragraphs of text spanning
165    multiple sentences.
166    """
167
168    def __init__(
169        self,
170        data_source: Sized,
171        target_size: int,
172        sample_range: Tuple[int, int],
173        num_jobs: int = 1,
174        seed: Optional[int] = None,
175    ):
176        """
177        Initialize a sub-sampler for language modelling data sets.
178
179        Parameters
180        ----------
181        data_source: Sized
182            Iterable of data corresponding to a split. Usually is a list of dicts, containing input ids, attention masks
183            and labels for each instance.
184        target_size: int
185            Number of instances that should be contained in the sub-sampled data set.
186        sample_range: Tuple[int, int]
187            Length of paragraphs that are sampled from the corpus. sample_range determines the ranges from which the
188            length is sampled uniformly.
189        num_jobs: int
190            Number of jobs used to process data before sampling.
191        seed: Optional[int]
192            Seed set for reproducibility.
193        """
194        self.length2instances = defaultdict(
195            lambda: []
196        )  # List of instances with the same sentence length
197        self.sample_range = sample_range
198        self.seq_length_freqs = defaultdict(int)  # Frequency of sentence lengths
199        super().__init__(data_source, target_size, num_jobs, seed)
200
201    def _create_indices(self, data_source: Sized):
202        """
203        Analyze the given data in order to determining the sampling strategy.
204
205        Parameters
206        ----------
207        data_source: Sized
208            Data containing instances of data split.
209        """
210
211        def _collect_stats(
212            i: int, instance: Dict[str, Any]
213        ) -> Tuple[Dict[int, Any], ...]:
214            seq_length = len(instance["input_ids"])
215
216            return (
217                {seq_length: 1},  # Into self.seq_length_freqs
218                {seq_length: [i]},  # Into self.length2instances
219            )
220
221        # Go through data and categorize it
222        parallel = Parallel(n_jobs=self.num_jobs)
223        stats = parallel(
224            delayed(_collect_stats)(i, data) for i, data in enumerate(data_source)
225        )
226
227        # Merge results
228        self.seq_length_freqs, self.length2instances = reduce(
229            lambda stats_a, stats_b: (
230                merge_freq_dicts(stats_a[0], stats_b[0]),
231                merge_instance_dicts(stats_a[1], stats_b[1]),
232            ),
233            stats,
234        )
235        del stats
236
237        # Compute probability of sampling an instance based on class and sentence length
238        instance_probs = np.zeros(len(data_source))
239        seq_length_probs = create_probs_from_dict(self.seq_length_freqs)
240
241        for seq_length in self.length2instances:
242
243            for i in self.length2instances[seq_length]:
244                # Probability for an instance to be sampled is the probability of the sequence length divided by the
245                # number of sequences with that same length
246                instance_probs[i] = seq_length_probs[seq_length] / len(
247                    self.length2instances[seq_length]
248                )
249
250        if self.seed is not None:
251            np.random.seed(self.seed)
252
253        self.indices = []
254
255        while len(self.indices) < self.target_size:
256            # Pick index
257            index = np.random.choice(
258                np.arange(len(data_source)),
259                size=1,
260                replace=False,
261                p=instance_probs,
262            )[0]
263
264            # Pick length
265            length = np.random.choice(np.arange(*self.sample_range), size=1)[0]
266            offset = min(
267                len(data_source) - 1, length, self.target_size - len(self.indices)
268            )  # Sampled paragraph should not flow over corpus boundaries or create more instances than specified
269
270            # Add the sampled indices
271            self.indices.extend(range(index, index + offset))
272
273            # Mask out the sampled indices so they cannot be sampled again and re-normalize the distribution
274            instance_probs[index : index + offset] = 0
275            instance_probs /= sum(instance_probs)
276
277        del self.seq_length_freqs, self.length2instances, seq_length_probs, instance_probs

280class SequenceClassificationSampler(Subsampler):
281    """
282    Sampler specific to sequence classification. The strategy here is to approximately maintain the same class
283    distribution as in the original corpus, and to a lesser extent the same sequence length distribution.
284    """
285
286    def __init__(
287        self,
288        data_source: Sized,
289        target_size: int,
290        num_jobs: int = 1,
291        seed: Optional[int] = None,
292    ):
293        """
294        Initialize a sub-sampler for sequence classification tasks.
295
296        Parameters
297        ----------
298        data_source: Sized
299            Iterable of data corresponding to a split. Usually is a list of dicts, containing input ids, attention masks
300            and labels for each instance.
301        target_size: int
302            Number of instances that should be contained in the sub-sampled data set.
303        num_jobs: int
304            Number of jobs used to process data before sampling.
305        seed: Optional[int]
306            Seed set for reproducibility.
307        """
308        # Dictionary mapping to a list instances of the same class and same sentence lengths
309        self.class2length2instances = defaultdict(lambda: defaultdict(lambda: []))
310        self.class_freqs = defaultdict(int)  # Frequency of classes
311        # Frequencies of sentence lengths of the same class
312        self.class_length_freqs = defaultdict(lambda: defaultdict(int))
313        super().__init__(data_source, target_size, num_jobs, seed)
314
315    def _create_indices(self, data_source: Sized):
316        """
317        Analyze the given data in order to determing the sampling strategy.
318
319        Parameters
320        ----------
321        data_source: Sized
322            Data containing instances of data split.
323        """
324
325        def _collect_stats(
326            i: int, instance: Dict[str, Any]
327        ) -> Tuple[Dict[int, Any], ...]:
328            label = instance["labels"].item()
329            seq_length = len(instance["input_ids"])
330
331            return (
332                {label: {seq_length: [i]}},  # Into self.class2length2instances
333                {label: 1},  # Into self.class_freqs
334                {label: {seq_length: 1}},  # Into self.class_length_freqs
335            )
336
337        # Go through data and categorize it
338        parallel = Parallel(n_jobs=self.num_jobs)
339        stats = parallel(
340            delayed(_collect_stats)(i, data) for i, data in enumerate(data_source)
341        )
342
343        # Merge results
344        # Since some of the dictionaries are nested, we also unfortunately have to apply nested merge operations
345        self.class2length2instances, self.class_freqs, self.class_length_freqs = reduce(
346            lambda stats_a, stats_b: (
347                # class2length2instances
348                {
349                    label: merge_instance_dicts(
350                        stats_a[0].get(label, {}), stats_b[0].get(label, {})
351                    )
352                    for label in set(stats_a[0].keys() | stats_b[0].keys())
353                },
354                # class_freqs
355                merge_freq_dicts(stats_a[1], stats_b[1]),
356                # class_length_freqs
357                {
358                    label: merge_freq_dicts(
359                        stats_a[2].get(label, {}), stats_b[2].get(label, {})
360                    )
361                    for label in set(stats_a[2].keys() | stats_b[2].keys())
362                },
363            ),
364            stats,
365        )
366
367        # Compute probability of sampling an instance based on class and sentence length
368        label_probs = create_probs_from_dict(self.class_freqs)
369        instance_probs = np.zeros(len(data_source))
370
371        for label in self.class_freqs:
372            seq_length_probs = create_probs_from_dict(self.class_length_freqs[label])
373
374            for seq_length in self.class2length2instances[label]:
375                for i in self.class2length2instances[label][seq_length]:
376                    # Probability for an instance to be sampled is the probability of the class label times the
377                    # probability of the sequence length given the class label divided by the number of sequences
378                    # with that same class and sequence length
379                    instance_probs[i] = (
380                        label_probs[label]
381                        * seq_length_probs[seq_length]
382                        / len(self.class2length2instances[label][seq_length])
383                    )
384
385        if self.seed is not None:
386            np.random.seed(self.seed)
387
388        self.indices = np.random.choice(
389            np.arange(len(data_source)),
390            size=min(self.target_size, len(data_source)),
391            replace=False,
392            p=instance_probs,
393        ).tolist()

396class TokenClassificationSampler(Subsampler):
397    """
398    Sampler specific to sequence classification. The strategy here is to approximately maintain the same class
399    distribution as in the original corpus, and to a lesser extent the same sequence length distribution. Compared to
400    the SequenceClassificationSampler, all labels of a sequence are considered.
401    """
402
403    def __init__(
404        self,
405        data_source: Sized,
406        target_size: int,
407        ignore_label: int = -100,
408        num_jobs: int = 1,
409        seed: Optional[int] = None,
410    ):
411        """
412        Initialize a sub-sampler for token classification tasks.
413
414        Parameters
415        ----------
416        data_source: Sized
417            Iterable of data corresponding to a split. Usually is a list of dicts, containing input ids, attention masks
418            and labels for each instance.
419        target_size: int
420            Number of instances that should be contained in the sub-sampled data set.
421        ignore_label: int
422            Determine label that should be ignored when computing input statistics used for sampling. Default is -100.
423        num_jobs: int
424            Number of jobs used to process data before sampling.
425        seed: Optional[int]
426            Seed set for reproducibility.
427        """
428        self.length2instances = defaultdict(
429            lambda: []
430        )  # List of instances with the same sentence length
431        self.seq_length_freqs = defaultdict(int)  # Frequency of sentence lengths
432        self.label_freqs = {}  # Frequencies of labels (over the whole corpus)
433        self.instance2label_freqs = {}  # Frequencies of labels (per instance)
434        self.ignore_label = ignore_label
435        super().__init__(data_source, target_size, num_jobs, seed)
436
437    def _create_indices(self, data_source: Sized):
438        """
439        Analyze the given data in order to determining the sampling strategy.
440
441        Parameters
442        ----------
443        data_source: Sized
444            Data containing instances of data split.
445        """
446
447        def _collect_stats(
448            i: int, instance: Dict[str, Any]
449        ) -> Tuple[Dict[int, Any], ...]:
450            seq_length = len(instance["input_ids"])
451            label_freqs = Counter(instance["labels"].tolist())
452            del label_freqs[self.ignore_label]
453
454            return (
455                {seq_length: 1},  # Into self.seq_length_freqs
456                {seq_length: [i]},  # Into self.length2instances
457                label_freqs,  # Into self.label_freqs
458                {i: label_freqs},  # Into self.instance2label_freqs
459            )
460
461        # Go through data and categorize it
462        parallel = Parallel(n_jobs=self.num_jobs)
463        stats = parallel(
464            delayed(_collect_stats)(i, data) for i, data in enumerate(data_source)
465        )
466
467        # Merge results
468        # Since some of the dictionaries are nested, we also unfortunately have to apply nested merge operations
469        (
470            self.seq_length_freqs,
471            self.length2instances,
472            self.label_freqs,
473            self.instance2label_freqs,
474        ) = reduce(
475            lambda stats_a, stats_b: (
476                # seq_length_freqs
477                merge_freq_dicts(stats_a[0], stats_b[0]),
478                # length2instances
479                merge_instance_dicts(stats_a[1], stats_b[1]),
480                # label_freqs
481                merge_freq_dicts(stats_a[2], stats_b[2]),
482                # instance2label_freqs
483                {**stats_a[3], **stats_b[3]},
484            ),
485            stats,
486        )
487
488        # Compute probability of sampling an instance based on class and sentence length
489        instance_probs = np.zeros(len(data_source))
490        seq_length_probs = create_probs_from_dict(self.seq_length_freqs)
491        label_probs = create_probs_from_dict(self.label_freqs)
492
493        for seq_length in self.length2instances:
494
495            log_lik = lambda p, q: sum(
496                p * np.log(q)
497            )  # Expected log-likelihood under corpus label distribution
498            # Create a probability distribution over instances of the sample length by computing the log-likelihood
499            # between the label distribution of an instance and the label distribution of a corpus and normalizing
500            # all the scores for one length "bucket" into the 0, 1 range. This way, sampling instances that match the
501            # overall label distribution becomes more likely.
502
503            # First, create distributions over labels per instances
504            seq_length_instance_probs = [
505                create_probs_from_dict(
506                    self.instance2label_freqs[i], max_label=len(label_probs) - 1
507                )
508                for i in self.length2instances[seq_length]
509            ]
510            # Secondly, compute cross-entropy scores
511            seq_length_instance_probs = np.array(
512                [
513                    log_lik(label_probs, label_dist + 1e-5)
514                    for label_dist in seq_length_instance_probs
515                ]
516            )
517            seq_length_instance_probs += min(
518                seq_length_instance_probs
519            )  # Make all values positive
520            seq_length_instance_probs /= sum(seq_length_instance_probs)  # Normalize
521
522            for i, instance_prob in zip(
523                self.length2instances[seq_length], seq_length_instance_probs
524            ):
525                # Probability for an instance to be sampled is the probability of the sequence length times the
526                # probability created by checking the frequencies of labels in a sequence with the overall labels in
527                # the corpus
528                instance_probs[i] = seq_length_probs[seq_length] * instance_prob
529
530        if self.seed is not None:
531            np.random.seed(self.seed)
532
533        self.indices = np.random.choice(
534            np.arange(len(data_source)),
535            size=min(self.target_size, len(data_source)),
536            replace=False,
537            p=instance_probs,
538        ).tolist()

433        self.label_freqs = {}  # Frequencies of labels (over the whole corpus)
434        self.instance2label_freqs = {}  # Frequencies of labels (per instance)

438    def _create_indices(self, data_source: Sized):
439        """
440        Analyze the given data in order to determining the sampling strategy.
441
442        Parameters
443        ----------
444        data_source: Sized
445            Data containing instances of data split.
446        """
447
448        def _collect_stats(
449            i: int, instance: Dict[str, Any]
450        ) -> Tuple[Dict[int, Any], ...]:
451            seq_length = len(instance["input_ids"])
452            label_freqs = Counter(instance["labels"].tolist())
453            del label_freqs[self.ignore_label]
454
455            return (
456                {seq_length: 1},  # Into self.seq_length_freqs
457                {seq_length: [i]},  # Into self.length2instances
458                label_freqs,  # Into self.label_freqs
459                {i: label_freqs},  # Into self.instance2label_freqs
460            )
461
462        # Go through data and categorize it
463        parallel = Parallel(n_jobs=self.num_jobs)
464        stats = parallel(
465            delayed(_collect_stats)(i, data) for i, data in enumerate(data_source)
466        )
467
468        # Merge results
469        # Since some of the dictionaries are nested, we also unfortunately have to apply nested merge operations
470        (
471            self.seq_length_freqs,
472            self.length2instances,
473            self.label_freqs,
474            self.instance2label_freqs,
475        ) = reduce(
476            lambda stats_a, stats_b: (
477                # seq_length_freqs
478                merge_freq_dicts(stats_a[0], stats_b[0]),
479                # length2instances
480                merge_instance_dicts(stats_a[1], stats_b[1]),
481                # label_freqs
482                merge_freq_dicts(stats_a[2], stats_b[2]),
483                # instance2label_freqs
484                {**stats_a[3], **stats_b[3]},
485            ),
486            stats,
487        )
488
489        # Compute probability of sampling an instance based on class and sentence length
490        instance_probs = np.zeros(len(data_source))
491        seq_length_probs = create_probs_from_dict(self.seq_length_freqs)
492        label_probs = create_probs_from_dict(self.label_freqs)
493
494        for seq_length in self.length2instances:
495
496            log_lik = lambda p, q: sum(
497                p * np.log(q)
498            )  # Expected log-likelihood under corpus label distribution
499            # Create a probability distribution over instances of the sample length by computing the log-likelihood
500            # between the label distribution of an instance and the label distribution of a corpus and normalizing
501            # all the scores for one length "bucket" into the 0, 1 range. This way, sampling instances that match the
502            # overall label distribution becomes more likely.
503
504            # First, create distributions over labels per instances
505            seq_length_instance_probs = [
506                create_probs_from_dict(
507                    self.instance2label_freqs[i], max_label=len(label_probs) - 1
508                )
509                for i in self.length2instances[seq_length]
510            ]
511            # Secondly, compute cross-entropy scores
512            seq_length_instance_probs = np.array(
513                [
514                    log_lik(label_probs, label_dist + 1e-5)
515                    for label_dist in seq_length_instance_probs
516                ]
517            )
518            seq_length_instance_probs += min(
519                seq_length_instance_probs
520            )  # Make all values positive
521            seq_length_instance_probs /= sum(seq_length_instance_probs)  # Normalize
522
523            for i, instance_prob in zip(
524                self.length2instances[seq_length], seq_length_instance_probs
525            ):
526                # Probability for an instance to be sampled is the probability of the sequence length times the
527                # probability created by checking the frequencies of labels in a sequence with the overall labels in
528                # the corpus
529                instance_probs[i] = seq_length_probs[seq_length] * instance_prob
530
531        if self.seed is not None:
532            np.random.seed(self.seed)
533
534        self.indices = np.random.choice(
535            np.arange(len(data_source)),
536            size=min(self.target_size, len(data_source)),
537            replace=False,
538            p=instance_probs,
